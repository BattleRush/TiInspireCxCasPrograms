Define LibPub custpoly(a,b)=
Prgm:zeronum:=zeros(a,x):zeronumlen:=dim(zeronum)::zeroden:=zeros(b,x):zerodenlen:=dim(zeroden)::Disp "NS Num Zeros",zeronum:Disp "NS Den Zeros",zeroden::Disp "NS All Zeros",zeros(((a)/(b)),x)::If zerodenlen=0 Then:    Disp "Def
 R \ { None }":    Else::    Disp "Def R \ ",zeroden::    fix:={}:    fixcount:=1:    For z,1,zerodenlen,1:    For t,1,zeronumlen,1:       If zeroden[z]=zeronum[t] Then:             fix[fixcount]:=zeroden[z]:                   fixcount:=fixcount+1:             EndIf:             EndFor:             EndFor::             nonfix:={}:             nonfixcount:=1:             For z,1,zerodenlen,1:             hit:=0:             For t,1,fixcount-1,1:                If zeroden[z]=fix[t] Then:                         hit:=1:                EndIf:                EndFor:                If hit=0 Then:                nonfix[nonfixcount]:=zeroden[z]:                nonfixcount:=nonfixcount+1:                EndIf:                EndFor:                Disp "Fixable",fix:                Disp "Pol ",nonfix:                EndIf::                asy:=polyQuotient(a,b):                If string(asy)="0" Then:                Disp "Asy x-Axis":                Else:                Disp "Asy",asy:                EndIf:EndPrgm
