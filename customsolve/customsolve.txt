Define LibPub customsolve(a,b,c)=
Prgm
:d:=b^(2)-4*a*c
:If d<0 Then
:  Disp "D=",d,char(10),"No solutions possible, because D is negative"
:Else
:
:  x1:=((−b-√(b^(2)-4*a*c))/(2*a))
:
:  x2:=((−b+√(b^(2)-4*a*c))/(2*a))
:
:  If x1≠x2 Then
:    Disp "D=",d,char(10),"x₁=",x1,char(10),"x₂=",x2
:  Else
:    Disp "D=",d,char(10),"x₁=",x1
:  EndIf
:
:EndIf
:EndPrgm
